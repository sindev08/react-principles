code Implementation

Pass the columns and data props to the Table component. 

import { Table } from '@design-system/core';

// Column definition
const columns = [
  { header: 'User', accessor: 'name', sortable: true },
  { header: 'Role', accessor: 'role' },
  { header: 'Status', accessor: 'status' }
];

function UserTable() {
  return (
    <Table 
      columns={columns} 
      data={userData}
      pagination={true}
      pageSize={10}
    />
  );
}
